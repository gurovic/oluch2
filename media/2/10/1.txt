def pr(a,b,k,n):
 reset()
 turn(180)
 for g in range(k):
  for v in range(n):
   for m in range(2):
    go(b)
    turn(-90)
    go(a)
    turn(-90)
   turn(-90)
   pen_up()
   go(a+b)
   turn(90)
   pen_down()
  turn(90)
  pen_up()
  go((a+b)*n)
  turn(-90)
  go(b*2)
  pen_down()


k-число рядов
n-число прямоугольников в ряду
a-длина прям.
b-ширина прям.
